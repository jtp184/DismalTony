def self.call(*args)
  return vi if args.empty?
  return vi.call(args[0]) if args.length == 1

  raise NoMethodError
end

def self.vi
  return config[:vi] if config[:vi]

  @@config[:vi] = DismalTony::VIBase.new(data_store: config[:data_store])
