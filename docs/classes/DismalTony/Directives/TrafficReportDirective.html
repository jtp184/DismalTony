def run
  x = determine_intent
  return x unless parameters[:info_type]

  check_for_start_and_end

  y = confirm_start_and_end
  confirmed = y.shift

  resp = if confirmed
           data_rep, r = choose_response

           return_data(data_rep)
           r
         else
           y.shift
  end

  resp
end
