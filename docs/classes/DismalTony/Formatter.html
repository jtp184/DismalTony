def self.format(str, opts = {})
  md = Formatter::INCOMING.match(str)
  em = (md['emote'] || 'smile')

  result = md['message']

  em = get_icon(em)
  em = opts[:use_icon] if opts[:use_icon]

  result = Formatter.add_icon(result, em) unless opts[:icon] == false
  result = extra_space(result) if opts[:extra_space]
  result
end
