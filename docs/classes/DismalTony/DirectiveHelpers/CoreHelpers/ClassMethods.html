def expect_frags(*these)
  @default_frags ||= {}
  these = these.first if these.one? && these.first.is_a?(Hash)

  case these
  when Array
    these.each { |a| @default_frags[a.to_sym] = nil }
  when Hash
    these.each_pair { |a, b| @default_frags[a.to_sym] = b }
  end
end
