def method_missing(method_name, *params, &blk)
  super unless parsed_results.any? { |r| r.respond_to?(method_name) }
  unless parsed_results.one? { |r| r.respond_to?(method_name) }
    raise NameError, "Multiple parsed results respond to ##{method_name}"
  end

  parsed_results.find { |r| r.respond_to?(method_name) }
                .method(method_name)
                .call(*params, &blk)
end
