def parse_web_req(resp)
  jsr = resp
  sym = jsr['Meta Data']['2. Symbol']
  slug = jsr.to_a[1].last

  results = []

  slug.each do |dat, prindx|
    md = /(\d+)-(\d+)-(\d+)/.match dat
    datum = Date.new(md[1].to_i, md[2].to_i, md[3].to_i)
    results << stock_price(sym, datum, *prindx.values.map(&:to_f))
  end
  results
end
